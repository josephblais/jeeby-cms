```javascript
async function handleSaveToLibrary() {
  if (!pendingMeta) return
  const title = pendingTitleRef.current?.value?.trim() ?? ''
  const alt = pendingAltRef.current?.value?.trim() ?? ''
  try {
    await addMediaItem(db, {
      storageUrl: pendingMeta.storageUrl,
      storagePath: pendingMeta.storagePath,
      title,
      alt,
      mimeType: pendingMeta.mimeType,
      size: pendingMeta.size,
    })
    // Now fill the block data with the uploaded URL and alt
    onChange({ ...data, src: pendingMeta.storageUrl, alt: alt || data?.alt || '' })
    setPendingMeta(null)
  } catch (err) {
    console.error('[jeeby-cms] Failed to save media item:', err)
  }
}

function handleSkipLibrary() {
  if (!pendingMeta) return
  // Fill block data without creating a library record
  onChange({ ...data, src: pendingMeta.storageUrl })
  setPendingMeta(null)
}
```

**Add library selection handler:**

```javascript
function handleLibrarySelect(item) {
  // item: { id, storageUrl, alt, title, storagePath, mimeType, size }
  setLibraryOpen(false)

  // Per D-10: if block alt is empty, auto-fill from library
  if (!data?.alt?.trim() && item.alt) {
    onChange({ ...data, src: item.storageUrl, alt: item.alt })
    return
  }

  // Per D-09: if block alt is non-empty and library alt differs, show conflict prompt
  if (data?.alt?.trim() && item.alt && item.alt !== data.alt) {
    onChange({ ...data, src: item.storageUrl })
    setAltConflict(item.alt)
    return
  }

  // No conflict -- just fill the URL
  onChange({ ...data, src: item.storageUrl })
}
```
