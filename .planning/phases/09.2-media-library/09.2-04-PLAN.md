```javascript
async function handleSaveToLibrary() {
  if (!pendingMeta) return
  const title = pendingTitleRef.current?.value?.trim() ?? ''
  const altInput = document.getElementById('pending-gallery-alt-' + blockId + '-' + index)
  const alt = altInput?.value?.trim() ?? ''
  try {
    await addMediaItem(db, {
      storageUrl: pendingMeta.storageUrl,
      storagePath: pendingMeta.storagePath,
      title,
      alt,
      mimeType: pendingMeta.mimeType,
      size: pendingMeta.size,
    })
    onChange({
      ...data,
      items: items.map((it, i) => i === index ? { ...it, src: pendingMeta.storageUrl, alt: alt || it.alt } : it),
    })
    setPendingMeta(null)
  } catch (err) {
    console.error('[jeeby-cms] Failed to save gallery media item:', err)
  }
}

function handleSkipLibrary() {
  if (!pendingMeta) return
  onChange({
    ...data,
    items: items.map((it, i) => i === index ? { ...it, src: pendingMeta.storageUrl } : it),
  })
  setPendingMeta(null)
}
```
